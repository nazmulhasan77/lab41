/****

-o localhost:7050 = address of the orderer node (ordering service).
 Overrides the hostname used for TLS verification.
--tls \ Enables TLS (Transport Layer Security) for secure communication.
Path to the TLS CA certificate of the orderer. Used to verify the orderer’s identity.
channel name → assetchannel.
chaincode (smart contract) → assetchaincode.
Address of peer0 of Org1.
TLS certificate of Org1 peer. 
Used to securely communicate with that peer.
Address of peer0 of Org2.
Used to securely communicate with that peer.

****/



# create new asset
peer chaincode invoke -o localhost:7050 \
  --ordererTLSHostnameOverride orderer.example.com \
  --tls \
  --cafile ${PWD}/organizations/ordererOrganizations/example.com/orderers/orderer.example.com/msp/tlscacerts/tlsca.example.com-cert.pem \
  -C assetchannel \
  -n assetchaincode \
  --peerAddresses localhost:7051 \
  --tlsRootCertFiles ${PWD}/organizations/peerOrganizations/org1.example.com/peers/peer0.org1.example.com/tls/ca.crt \
  --peerAddresses localhost:9051 \
  --tlsRootCertFiles ${PWD}/organizations/peerOrganizations/org2.example.com/peers/peer0.org2.example.com/tls/ca.crt \
  -c '{"function":"CreateAsset","Args":["asset2","Car","15000","Sakib"]}'
